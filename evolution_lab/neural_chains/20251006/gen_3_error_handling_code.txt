#include <iostream>
#include <exception>

/**
 * Hello World with Error Handling
 * Generation: 1
 * STL Patterns: iostream, exception
 */

int main() {
    try {
        std::cout << "Hello, World!" << std::endl;
        return 0;
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
}
